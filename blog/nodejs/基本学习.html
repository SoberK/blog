# 查看信息

fs.stat('./test.txt', (err, stats) => { 
    console.log(stats)
    console.log(stats.isFile())  //是否是文件
    console.log(stats.isDirectory()) //是否是文件目录
    console.log(stats.isSocket()) //是否是网络文件
})

# express框架

# 初体验

const express = require('express');
const app = express()

app.get('/', function (req, res) {
    res.end('hello')
})

app.listen(3000, e=>{
    console.log(e)
})

# 获取报文

  • 请求req:
    • method
    • url
    • httpVersion
    • path
    • query
    • ip
    • params
  • 响应res:
    • status(500)
    • set(name,value)
    • send(data)
    • redirect(url)
    • download(url)
    • json() 响应json
    • sendFile(path) 响应文件的功能
